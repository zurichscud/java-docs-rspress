# 请求日志过滤器

## 功能点

- 请求参数RequestParam的打印：QueryString、FormData
- 请求体RequestBody的打印
- 表单中的文件数据需要打印文件名，而不是二进制数据
- 表单中密码字段的脱敏处理
- 需要解决重复读取请求体的问题

打印时可分为params/body（json）/files三个字段场景

## 实现

```java
@Slf4j
@Component
@Order(Ordered.HIGHEST_PRECEDENCE + 1)
public class RequestLogFilter extends OncePerRequestFilter {

    private static final Set<String> PASSWORD_FIELDS = Set.of(
            "password",
            "pwd",
            "oldPassword",
            "newPassword"
    );

    private final ObjectMapper objectMapper = new ObjectMapper();

    @Override
    protected void doFilterInternal(
            HttpServletRequest request,
            HttpServletResponse response,
            FilterChain filterChain)
            throws ServletException, IOException {

        long start = System.currentTimeMillis();

        try {
            filterChain.doFilter(request, response);
        } finally {

            logRequest(request, System.currentTimeMillis() - start);
        }
    }

    private void logRequest(HttpServletRequest request, long cost) {

        String method = request.getMethod();
        String uri = request.getRequestURI();

        String queryString = buildQueryString(request);

        String body = buildRequestBody(request);

        log.info("""
                        
                ==============================
                Request URI   : {}
                Method        : {}
                QueryString   : {}
                RequestBody   : {}
                Cost          : {} ms
                ==============================
                """,
                uri,
                method,
                queryString,
                body,
                cost);
    }

    private String buildQueryString(HttpServletRequest request) {

        Map<String, String[]> map = request.getParameterMap();

        if (map.isEmpty()) {
            return "";
        }

        Map<String, Object> result = new LinkedHashMap<>();

        map.forEach((k, v) -> {
            if (PASSWORD_FIELDS.contains(k)) {
                result.put(k, "******");
            } else {
                result.put(k, Arrays.asList(v));
            }
        });

        try {
            return objectMapper.writeValueAsString(result);
        } catch (Exception e) {
            return result.toString();
        }
    }

    private String buildRequestBody(HttpServletRequest request) {

        if (!(request instanceof ContentCachingRequestWrapper wrapper)) {
            return "";
        }

        String contentType = request.getContentType();

        try {

            if (contentType != null &&
                contentType.startsWith(MediaType.MULTIPART_FORM_DATA_VALUE)) {

                return buildMultipartBody(request);
            }

            byte[] content = wrapper.getContentAsByteArray();

            if (content.length == 0) {
                return "";
            }

            String body = new String(
                    content,
                    wrapper.getCharacterEncoding() == null
                            ? StandardCharsets.UTF_8
                            : Charset.forName(wrapper.getCharacterEncoding())
            );

            return maskJsonPassword(body);

        } catch (Exception e) {
            return "[parse body failed]";
        }
    }

    private String buildMultipartBody(HttpServletRequest request)
            throws Exception {

        if (!(request instanceof MultipartHttpServletRequest multipart)) {
            return "";
        }

        Map<String, Object> result = new LinkedHashMap<>();

        // 普通表单字段
        multipart.getParameterMap().forEach((k, v) -> {

            if (PASSWORD_FIELDS.contains(k)) {
                result.put(k, "******");
            } else {
                result.put(k, Arrays.asList(v));
            }
        });

        // 文件字段
        multipart.getFileMap().forEach((name, file) -> {

            result.put(name, Map.of(
                    "fileName", file.getOriginalFilename(),
                    "size", file.getSize()
            ));
        });

        return objectMapper.writeValueAsString(result);
    }

    private String maskJsonPassword(String body) {

        try {

            JsonNode root = objectMapper.readTree(body);

            maskNode(root);

            return objectMapper.writeValueAsString(root);

        } catch (Exception e) {

            return body;
        }
    }

    private void maskNode(JsonNode node) {

        if (node instanceof ObjectNode objectNode) {

            Iterator<String> fields = objectNode.fieldNames();

            while (fields.hasNext()) {

                String field = fields.next();

                JsonNode child = objectNode.get(field);

                if (PASSWORD_FIELDS.contains(field)) {
                    objectNode.put(field, "******");
                } else {
                    maskNode(child);
                }
            }
        }

        if (node instanceof ArrayNode arrayNode) {
            arrayNode.forEach(this::maskNode);
        }
    }
}
```
